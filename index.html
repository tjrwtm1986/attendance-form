<script>

// ── CONFIG — paste your Apps Script Web App URL here ──────────

const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbx7LvEgu1N6kGqlfyLtCNC5Da5we4zcquWPKNP-us4qRU86tH6QtmYQv6EgKPpaRSop/exec";

// ─────────────────────────────────────────────────────────────


let gpsData   = null;   // { latitude, longitude, accuracy }

let photoData = null;   // base64 JPEG string

let stream    = null;   // MediaStream


// ── GPS ───────────────────────────────────────────────────────

function getLocation() {

  if (!navigator.geolocation) {

    setGps("err", "Geolocation not supported by this browser.");

    return;

  }

  setGps("busy", "Detecting your location…");

  navigator.geolocation.getCurrentPosition(

    pos => {

      gpsData = {

        latitude:  pos.coords.latitude.toFixed(6),

        longitude: pos.coords.longitude.toFixed(6),

        accuracy:  Math.round(pos.coords.accuracy)

      };

      setGps("ok", `📍 ${gpsData.latitude}, ${gpsData.longitude}  (±${gpsData.accuracy} m)`);

      checkReady();

    },

    err => {

      const msgs = {

        1: "Location access denied. Please allow location in your browser settings, then reload.",

        2: "Location unavailable. Move to an area with better signal.",

        3: "Location request timed out. Please reload and try again."

      };

      setGps("err", msgs[err.code] || "Location error.");

    },

    { enableHighAccuracy: true, timeout: 15000, maximumAge: 0 }

  );

}


function setGps(state, text) {

  const strip = document.getElementById("gpsStrip");

  strip.className = "gps-strip " + state;

  document.getElementById("gpsText").textContent = text;

}


// ── Camera ───────────────────────────────────────────────────

async function startCamera() {

  const btn = document.getElementById("startCamBtn");

  btn.disabled = true;

  btn.textContent = "Starting…";

  try {

    stream = await navigator.mediaDevices.getUserMedia({

      video: { facingMode: "user", width: { ideal: 640 }, height: { ideal: 640 } },

      audio: false

    });

    const video = document.getElementById("videoEl");

    video.srcObject = stream;

    document.getElementById("camOverlay").style.display = "none";

    document.getElementById("camBtnRow").innerHTML = `

      <button class="cam-btn primary" onclick="capturePhoto()">📸 Take Selfie</button>

    `;

  } catch (e) {

    btn.disabled = false;

    btn.textContent = "Start Camera";

    setStatus("err", "Camera access denied. Please allow camera and reload.");

  }

}


function capturePhoto() {

  const video  = document.getElementById("videoEl");

  const canvas = document.getElementById("canvasEl");

  const size   = Math.min(video.videoWidth, video.videoHeight);

  canvas.width  = size;

  canvas.height = size;

  const ctx = canvas.getContext("2d");

  // Centre-crop to square

  const sx = (video.videoWidth  - size) / 2;

  const sy = (video.videoHeight - size) / 2;

  ctx.drawImage(video, sx, sy, size, size, 0, 0, size, size);

  photoData = canvas.toDataURL("image/jpeg", 0.8);


  // Show preview, stop stream

  const preview = document.getElementById("previewImg");

  preview.src = photoData;

  preview.style.display = "block";

  video.style.display = "none";

  stopStream();


  document.getElementById("camBtnRow").innerHTML = `

    <button class="cam-btn retake" onclick="retakePhoto()">↩ Retake</button>

  `;

  checkReady();

}


function retakePhoto() {

  photoData = null;

  const preview = document.getElementById("previewImg");

  preview.style.display = "none";

  preview.src = "";

  const video = document.getElementById("videoEl");

  video.style.display = "block";

  document.getElementById("camBtnRow").innerHTML = `

    <button class="cam-btn primary" onclick="startCamera()">Start Camera</button>

  `;

  document.getElementById("camOverlay").style.display = "flex";

  checkReady();

}


function stopStream() {

  if (stream) { stream.getTracks().forEach(t => t.stop()); stream = null; }

}


// ── Validation ───────────────────────────────────────────────

function checkReady() {

  const ok =

    document.getElementById("clockType").value &&

    document.getElementById("workerName").value &&

    document.getElementById("outlet").value &&

    gpsData && photoData;

  document.getElementById("submitBtn").disabled = !ok;

}


// Run checkReady whenever a dropdown changes

document.addEventListener("DOMContentLoaded", () => {

  ["clockType","workerName","outlet"].forEach(id =>

    document.getElementById(id).addEventListener("change", checkReady)

  );

  getLocation();

});


// ── Submit ────────────────────────────────────────────────────

async function submitForm() {

  if (!gpsData || !photoData) return;


  const btn = document.getElementById("submitBtn");

  btn.disabled = true;

  btn.textContent = "Submitting…";

  setStatus("", "Uploading — please wait…");


  const payload = {

    timestamp: new Date().toLocaleString("en-SG", { timeZone: "Asia/Singapore" }),

    name:      document.getElementById("workerName").value,

    outlet:    document.getElementById("outlet").value,

    clockType: document.getElementById("clockType").value,

    latitude:  gpsData.latitude,

    longitude: gpsData.longitude,

    accuracy:  gpsData.accuracy,

    photo:     photoData

  };


  try {

    const res  = await fetch(SCRIPT_URL, {

      method: "POST",

      body: JSON.stringify(payload)

    });

    const json = await res.json();


    if (json.success) {

      showSuccess(payload);

    } else {

      throw new Error(json.error || "Unknown error");

    }

  } catch (err) {

    setStatus("err", "Submission failed: " + err.message);

    btn.disabled = false;

    btn.textContent = "Submit Attendance";

  }

}


function showSuccess(p) {

  document.getElementById("formBody").style.display    = "none";

  document.getElementById("successScreen").style.display = "block";

  document.getElementById("successDetails").innerHTML = `

    <strong>Name:</strong> ${p.name}<br>

    <strong>Outlet:</strong> ${p.outlet}<br>

    <strong>Type:</strong> ${p.clockType}<br>

    <strong>Time:</strong> ${p.timestamp}<br>

    <strong>Location:</strong> ${p.latitude}, ${p.longitude} (±${p.accuracy} m)

  `;

}


function resetForm() {

  document.getElementById("formBody").style.display    = "flex";

  document.getElementById("successScreen").style.display = "none";

  ["clockType","workerName","outlet"].forEach(id =>

    document.getElementById(id).value = ""

  );

  photoData = null;

  const preview = document.getElementById("previewImg");

  preview.style.display = "none";

  preview.src = "";

  document.getElementById("videoEl").style.display = "block";

  document.getElementById("camOverlay").style.display = "flex";

  document.getElementById("camBtnRow").innerHTML =

    `<button class="cam-btn primary" id="startCamBtn" onclick="startCamera()">Start Camera</button>`;

  document.getElementById("submitBtn").disabled = true;

  document.getElementById("submitBtn").textContent = "Submit Attendance";

  setStatus("", "");

  getLocation();

}


function setStatus(type, msg) {

  const el = document.getElementById("statusMsg");

  el.className = type;

  el.textContent = msg;

}
